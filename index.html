// Замени весь блок управления (EventListeners) на этот: let touchStartX = null; window.addEventListener('touchstart', (e) => { // Если тап в правой части экрана — запоминаем начало свайпа if (e.touches[0].clientX > window.innerWidth / 2) { touchStartX = e.touches[0].clientX; } }, {passive: false}); window.addEventListener('touchmove', (e) => { // КРИТИЧНО: Запрещаем браузеру двигать страницу if (e.cancelable) e.preventDefault(); if (joyPos.active) { let rect = joyBox.getBoundingClientRect(); let touch = Array.from(e.touches).find(t => t.clientX < window.innerWidth / 2) || e.touches[0]; let dx = touch.clientX - (rect.left + 50); let dy = touch.clientY - (rect.top + 50); let angle = Math.atan2(dy, dx); let dist = Math.min(Math.sqrt(dx*dx + dy*dy), 40); joyPos.x = Math.cos(angle) * (dist / 40); joyPos.y = Math.sin(angle) * (dist / 40); stick.style.transform = `translate(${joyPos.x * 30}px, ${joyPos.y * 30}px)`; } // Плавный поворот камеры (справа) for (let t of e.touches) { if (t.clientX > window.innerWidth / 2) { if (touchStartX !== null) { let diffX = t.clientX - touchStartX; p.a += diffX * 0.007; // Чувствительность поворота touchStartX = t.clientX; } } } }, {passive: false}); window.addEventListener('touchend', (e) => { if (e.touches.length === 0) { joyPos.active = false; joyPos.x = 0; joyPos.y = 0; stick.style.transform = `translate(0,0)`; touchStartX = null; } });
